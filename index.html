      ✅ 桌機：滑鼠移動＝準星；點擊＝射擊<br/>

      📷 背景：把你的景點真實照片命名為 <b>bg.jpg</b> 放同一資料夾即可

  <audio id="shootSfx" src="shoot.wav" preload="auto"></audio>

  <audio id="hitSfx" src="hit.wav" preload="auto"></audio>

 

<script>

(() => {

  const canvas = document.getElementById("c");

  const ctx = canvas.getContext("2d", { alpha: false });

 

  const scoreEl = document.getElementById("score");

  const hpEl = document.getElementById("hp");

  const comboEl = document.getElementById("combo");

  const pauseBtn = document.getElementById("pauseBtn");

  const restartBtn = document.getElementById("restartBtn");

  const shootSfx = document.getElementById("shootSfx");

  const hitSfx = document.getElementById("hitSfx");

 

  // ====== 背景：真實景點照片（換成你的檔名/網址都可以） ======

  const bgImg = new Image();

  bgImg.src = "bg.jpg"; // 或改成完整網址

  let bgReady = false;

  bgImg.onload = () => bgReady = true;

 

  // ====== 遊戲參數（可微調） ======

  const config = {

    targetSpawnMs: 650,      // 目標生成間隔（越小越難）

    targetSpeedMin: 40,

    targetSpeedMax: 110,

    bulletSpeed: 720,

    targetRadius: 22,

    bulletRadius: 5,

    maxTargets: 10,

    hp: 3,

    // 背景顯示方式：cover（裁切鋪滿） or contain（完整但可能留黑邊）

    bgMode: "cover",

  };

 

  // ====== 狀態 ======

  let W = 0, H = 0, DPR = 1;

  let running = true;

  let lastT = performance.now();

  let lastSpawn = 0;

 

  let score = 0;

  let hp = config.hp;

  let combo = 0;

 

  const player = { x: 0, y: 0 }; // 準星位置

  const bullets = [];

  const targets = [];

  const particles = [];

 

  function resize() {

    DPR = Math.max(1, Math.min(2, window.devicePixelRatio || 1));

    W = Math.floor(window.innerWidth);

    H = Math.floor(window.innerHeight);

    canvas.width = Math.floor(W * DPR);

    canvas.height = Math.floor(H * DPR);

    ctx.setTransform(DPR, 0, 0, DPR, 0, 0);

    player.x = W * 0.5;

    player.y = H * 0.65;

  }

  window.addEventListener("resize", resize, { passive: true });

  resize();

 

  // ====== 輔助：背景 cover/contain 繪製 ======

  function drawBackground() {

    if (!bgReady) {

      // 背景尚未載入：用漸層替代

      const g = ctx.createLinearGradient(0, 0, 0, H);

      g.addColorStop(0, "#0b0c10");

      g.addColorStop(1, "#141622");

      ctx.fillStyle = g;

      ctx.fillRect(0, 0, W, H);

      return;

    }

 

    const iw = bgImg.naturalWidth, ih = bgImg.naturalHeight;

    if (!iw || !ih) return;

 

    const sx = 0, sy = 0;

    let sw = iw, sh = ih;

    let dx = 0, dy = 0, dw = W, dh = H;

 

    const ir = iw / ih;

    const cr = W / H;

 

    if (config.bgMode === "cover") {

      // 裁切鋪滿

      if (ir > cr) {

        // 圖太寬，裁左右

        sh = ih;

        sw = Math.floor(ih * cr);

        const cut = Math.floor((iw - sw) / 2);

        ctx.drawImage(bgImg, cut, 0, sw, sh, 0, 0, W, H);

      } else {

        // 圖太高，裁上下

        sw = iw;

        sh = Math.floor(iw / cr);

        const cut = Math.floor((ih - sh) / 2);

        ctx.drawImage(bgImg, 0, cut, sw, sh, 0, 0, W, H);

      }

    } else {

      // contain：完整顯示，可能留黑邊

      if (ir > cr) {

        dw = W;

        dh = Math.floor(W / ir);

        dx = 0;

        dy = Math.floor((H - dh) / 2);

      } else {

        dh = H;

        dw = Math.floor(H * ir);

        dy = 0;

        dx = Math.floor((W - dw) / 2);

      }

      ctx.fillStyle = "#0b0c10";

      ctx.fillRect(0, 0, W, H);

      ctx.drawImage(bgImg, sx, sy, sw, sh, dx, dy, dw, dh);

    }

 

    // 輕微暗角讓目標更清楚

    const vg = ctx.createRadialGradient(W*0.5, H*0.5, Math.min(W,H)*0.1, W*0.5, H*0.5, Math.max(W,H)*0.65);

    vg.addColorStop(0, "rgba(0,0,0,0)");

    vg.addColorStop(1, "rgba(0,0,0,0.45)");

    ctx.fillStyle = vg;

    ctx.fillRect(0, 0, W, H);

  }

 

  // ====== 生成目標 ======

  function spawnTarget() {

    if (targets.length >= config.maxTargets) return;

 

    const side = Math.floor(Math.random() * 3); // 0:上, 1:左, 2:右

    let x, y, vx, vy;

 

    const speed = rand(config.targetSpeedMin, config.targetSpeedMax);

    if (side === 0) {

      x = rand(30, W - 30); y = -30;

      vx = rand(-30, 30); vy = speed;

    } else if (side === 1) {

      x = -30; y = rand(60, H * 0.6);

      vx = speed; vy = rand(-20, 40);

    } else {

      x = W + 30; y = rand(60, H * 0.6);

      vx = -speed; vy = rand(-20, 40);

    }

 

    targets.push({

      x, y, vx, vy,

      r: config.targetRadius,

      hp: 1,

      t: 0,

    });

  }

 

  // ====== 射擊 ======

  function shoot(tx, ty) {

    if (!running) return;

    // 音效（若檔案不存在，catch 也沒事）

    try { shootSfx.currentTime = 0; shootSfx.play().catch(()=>{}); } catch(e){}

 

    const x = player.x, y = player.y;

    const dx = tx - x, dy = ty - y;

    const len = Math.hypot(dx, dy) || 1;

    const vx = (dx / len) * config.bulletSpeed;

    const vy = (dy / len) * config.bulletSpeed;

 

    bullets.push({ x, y, vx, vy, r: config.bulletRadius, life: 1.1 });

  }

 

  // ====== 碰撞 ======

  function hitTestCircle(a, b) {

    const dx = a.x - b.x, dy = a.y - b.y;

    const rr = a.r + b.r;

    return (dx*dx + dy*dy) <= rr*rr;

  }

 

  // ====== 粒子 ======

  function boom(x, y) {

    for (let i = 0; i < 14; i++) {

      const ang = rand(0, Math.PI * 2);

      const sp = rand(80, 260);

      particles.push({

        x, y,

        vx: Math.cos(ang) * sp,

        vy: Math.sin(ang) * sp,

        life: rand(0.25, 0.6),

      });

    }

  }

 

  // ====== 更新/繪製 ======

  function update(dt) {

    // 生成目標

    lastSpawn += dt * 1000;

    if (lastSpawn >= config.targetSpawnMs) {

      lastSpawn = 0;

      spawnTarget();

    }

 

    // 子彈

    for (let i = bullets.length - 1; i >= 0; i--) {

      const b = bullets[i];

      b.x += b.vx * dt;

      b.y += b.vy * dt;

      b.life -= dt;

      if (b.life <= 0 || b.x < -50 || b.x > W+50 || b.y < -50 || b.y > H+50) {

        bullets.splice(i, 1);

      }

    }

 

    // 目標

    for (let i = targets.length - 1; i >= 0; i--) {

      const t = targets[i];

      t.t += dt;

      t.x += t.vx * dt;

      t.y += t.vy * dt;

 

      // 目標掉到下方算失誤扣血

      if (t.y > H + 60) {

        targets.splice(i, 1);

        combo = 0;

        hp -= 1;

        if (hp <= 0) gameOver();

      }

    }

 

    // 碰撞：子彈 vs 目標

    for (let i = bullets.length - 1; i >= 0; i--) {

      const b = bullets[i];

      for (let j = targets.length - 1; j >= 0; j--) {

        const t = targets[j];

        if (hitTestCircle(b, t)) {

          bullets.splice(i, 1);

          targets.splice(j, 1);

          combo += 1;

          score += 10 + Math.min(20, combo * 2);

          boom(t.x, t.y);

          try { hitSfx.currentTime = 0; hitSfx.play().catch(()=>{}); } catch(e){}

          break;

        }

      }

    }

 

    // 粒子

    for (let i = particles.length - 1; i >= 0; i--) {

      const p = particles[i];

      p.x += p.vx * dt;

      p.y += p.vy * dt;

      p.vx *= Math.pow(0.05, dt);

      p.vy *= Math.pow(0.05, dt);

      p.life -= dt;

      if (p.life <= 0) particles.splice(i, 1);

    }

 

    // UI

    scoreEl.textContent = String(score);

    hpEl.textContent = String(hp);

    comboEl.textContent = String(combo);

  }

 

  function draw() {

    drawBackground();

 

    // 目標

    for (const t of targets) {

      ctx.beginPath();

      ctx.arc(t.x, t.y, t.r, 0, Math.PI*2);

      ctx.fillStyle = "rgba(255, 230, 0, 0.92)";

      ctx.fill();

      ctx.lineWidth = 2;

      ctx.strokeStyle = "rgba(0,0,0,0.35)";

      ctx.stroke();

    }

 

    // 子彈

    for (const b of bullets) {

      ctx.beginPath();

      ctx.arc(b.x, b.y, b.r, 0, Math.PI*2);

      ctx.fillStyle = "rgba(255,255,255,0.95)";

      ctx.fill();

    }

 

    // 粒子

    for (const p of particles) {

      ctx.beginPath();

      ctx.arc(p.x, p.y, 2.2, 0, Math.PI*2);

      ctx.fillStyle = "rgba(255, 255, 255, 0.75)";

      ctx.fill();

    }

 

    // 準星

    ctx.save();

    ctx.translate(player.x, player.y);

    ctx.strokeStyle = "rgba(255,255,255,0.9)";

    ctx.lineWidth = 2;

    ctx.beginPath();

    ctx.arc(0, 0, 18, 0, Math.PI*2);

    ctx.stroke();

    ctx.beginPath();

    ctx.moveTo(-24, 0); ctx.lineTo(-8, 0);

    ctx.moveTo(8, 0); ctx.lineTo(24, 0);

    ctx.moveTo(0, -24); ctx.lineTo(0, -8);

    ctx.moveTo(0, 8); ctx.lineTo(0, 24);

    ctx.stroke();

    ctx.restore();

 

    if (!running) {

      ctx.fillStyle = "rgba(0,0,0,0.55)";

      ctx.fillRect(0, 0, W, H);

      ctx.fillStyle = "#fff";

      ctx.font = "900 28px system-ui";

      ctx.fillText("遊戲結束", 24, 56);

      ctx.font = "700 16px system-ui";

      ctx.fillText(`分數：${score}（點「重來」再玩一次）`, 24, 84);

    }

  }

 

  function loop(now) {

    const dt = Math.min(0.033, (now - lastT) / 1000);

    lastT = now;

 

    if (running) update(dt);

    draw();

 

    requestAnimationFrame(loop);

  }

 

  function gameOver() {

    running = false;

    pauseBtn.textContent = "繼續";

  }

 

  function restart() {

    score = 0;

    hp = config.hp;

    combo = 0;

    bullets.length = 0;

    targets.length = 0;

    particles.length = 0;

    running = true;

    pauseBtn.textContent = "暫停";

  }

 

  function rand(a, b){ return a + Math.random() * (b - a); }

 

  // ====== 操作（滑鼠/觸控） ======

  function setAim(clientX, clientY) {

    player.x = Math.max(0, Math.min(W, clientX));

    player.y = Math.max(0, Math.min(H, clientY));

  }

 

  canvas.addEventListener("mousemove", (e) => setAim(e.clientX, e.clientY), { passive:true });

  canvas.addEventListener("click", (e) => { setAim(e.clientX, e.clientY); shoot(e.clientX, e.clientY); });

 

  canvas.addEventListener("touchstart", (e) => {

    const t = e.touches[0];

    if (!t) return;

    setAim(t.clientX, t.clientY);

    shoot(t.clientX, t.clientY);

  }, { passive:true });

 

  canvas.addEventListener("touchmove", (e) => {

    const t = e.touches[0];

    if (!t) return;

    setAim(t.clientX, t.clientY);

  }, { passive:true });

 

  // ====== UI Buttons ======

  pauseBtn.addEventListener("click", () => {

    running = !running;

    pauseBtn.textContent = running ? "暫停" : "繼續";

  });

  restartBtn.addEventListener("click", restart);

 

  // ====== 啟動 ======

  requestAnimationFrame(loop);

})();

</script>
